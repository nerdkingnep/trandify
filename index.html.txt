                        <tr key={v.id} className="hover:bg-blue-50/20 cursor-pointer transition-colors group" onClick={() => window.open(`https://www.youtube.com/watch?v=${v.id}`, '_blank')}>
                          <td className="p-4 md:p-6 text-center font-black text-gray-300 text-xl md:text-2xl group-hover:text-blue-500 transition-colors">{i+1}</td>
                          <td className="p-4 md:p-6">
                            <div className="relative">
                              <img src={v.snippet.thumbnails.medium.url} className="rounded-lg md:rounded-xl w-32 h-18 md:w-40 md:h-24 object-cover shadow-sm border border-gray-100 group-hover:shadow-md transition-shadow" />
                            </div>
                          </td>
                          <td className="p-4 md:p-6">
                            <p className="font-bold text-sm md:text-base line-clamp-2 mb-2 md:mb-3 text-gray-800 group-hover:text-blue-700 transition-colors leading-snug">{v.snippet.title}</p>
                            <div className="mb-2 md:mb-3">
                              <span className="text-[10px] md:text-xs font-black text-indigo-600 bg-indigo-50 border border-indigo-100 px-2 py-1 rounded-md">#{getMainKeyword(v)}</span>
                            </div>
                            <div className="flex flex-wrap items-center gap-2 md:gap-3">
                                <span className="font-bold text-[10px] md:text-[11px] text-blue-700 bg-blue-50 border border-blue-100 px-2 md:px-2.5 py-0.5 md:py-1 rounded-md">{v.snippet.channelTitle}</span>
                                <span className="text-[10px] md:text-[11px] font-bold text-gray-400">업로드: {formatDate(v.snippet.publishedAt)}</span>
                            </div>
                          </td>
                          <td className="p-4 md:p-6 text-right pr-6 md:pr-10">
                            <div className="flex flex-col items-end gap-1 md:gap-1.5">
                              <div className="flex items-center justify-end w-full"><span className="text-[9px] md:text-[10px] text-gray-400 font-bold mr-2 md:mr-3 uppercase tracking-wider">조회수</span><span className="font-black text-base md:text-lg text-gray-900 w-14 md:w-16 text-right">{formatShortNumber(v.statistics.viewCount)}</span></div>
                              <div className="flex items-center justify-end w-full"><span className="text-[9px] md:text-[10px] text-gray-400 font-bold mr-2 md:mr-3 uppercase tracking-wider">좋아요</span><span className="font-bold text-sm md:text-base text-gray-600 w-14 md:w-16 text-right">{formatShortNumber(v.statistics.likeCount)}</span></div>
                              <div className="flex items-center justify-end w-full"><span className="text-[9px] md:text-[10px] text-gray-400 font-bold mr-2 md:mr-3 uppercase tracking-wider">댓글</span><span className="font-bold text-sm md:text-base text-gray-600 w-14 md:w-16 text-right">{formatShortNumber(v.statistics.commentCount)}</span></div>
                            </div>
                          </td>
                        </tr>
                      ))}
                    </tbody>
                  </table>
                </div>
              }
            </div>
          )}

          {activeTab === 'growth' && (
            <div className="space-y-6">
              <div className="bg-white rounded-2xl md:rounded-[32px] p-6 md:p-10 border border-gray-100 flex flex-col xl:flex-row items-start xl:items-center justify-between gap-4 md:gap-6 shadow-sm bg-gradient-to-br from-white to-gray-50/50">
                <div>
                  <h3 className="text-xl md:text-2xl font-black text-gray-900 tracking-tight flex items-center gap-2">
                    콘텐츠 성장 시뮬레이터 <span className="text-[10px] md:text-xs bg-indigo-100 text-indigo-600 px-1.5 md:px-2 py-0.5 md:py-1 rounded-md font-black">개발 중</span>
                  </h3>
                  <p className="text-xs md:text-sm text-gray-500 mt-1 md:mt-2 font-medium">분석된 영상들의 현재 조회수를 스냅샷으로 기록하여 시간 경과에 따른 성장폭을 계산합니다.</p>
                </div>
                
                <div className="flex flex-col sm:flex-row items-stretch sm:items-center gap-3 md:gap-4 w-full xl:w-auto">
                  <select value={selectedGrowthChannel} onChange={(e) => setSelectedGrowthChannel(e.target.value)} className="border border-gray-200 bg-white rounded-xl px-4 py-3 md:py-4 text-xs md:text-sm font-bold text-gray-700 shadow-sm outline-none cursor-pointer w-full sm:w-auto">
                    <option value="all">전체 채널 보기</option>
                    {channels.map(c => <option key={c.id} value={c.id}>{c.title}</option>)}
                  </select>
                  <button onClick={saveGrowthRecord} disabled={savingGrowth || !dashboardVideos.length} className="bg-gray-900 hover:bg-black text-white px-6 md:px-10 py-3 md:py-4 rounded-xl md:rounded-2xl font-black flex justify-center items-center gap-2 md:gap-3 transition-all shadow-xl active:scale-95 disabled:opacity-50 text-sm md:text-base w-full sm:w-auto">
                    {savingGrowth ? <Loader2 size={20} className="animate-spin md:w-6 md:h-6"/> : <Save size={20} className="md:w-6 md:h-6"/>} 현재 스냅샷 저장
                  </button>
                </div>
              </div>

              {!filteredGrowthRecords.length ? (
                <div className="bg-white p-20 md:p-40 flex flex-col items-center justify-center text-center border-2 border-dashed border-gray-100 rounded-2xl md:rounded-[40px]">
                  <Activity className="w-10 h-10 md:w-12 md:h-12 text-gray-200 mb-4 md:mb-6"/>
                  <h4 className="text-lg md:text-xl font-black text-gray-400">조건에 맞는 성장 기록이 없습니다</h4>
                  <p className="text-xs md:text-sm font-medium text-gray-400 mt-2">성과 분석 탭에서 데이터를 불러온 뒤 스냅샷을 저장하거나 다른 채널을 선택해 보세요.</p>
                </div>
              ) : (
                <div className="grid grid-cols-1 gap-4 md:gap-6">
                  {filteredGrowthRecords.map(record => {
                    const expanded = expandedVideoId === record.id;
                    const keys = Object.keys(record.history).sort((a,b)=>new Date(a)-new Date(b));
                    const latest = record.history[keys[keys.length-1]];
                    return (
                      <div key={record.id} className={`bg-white border rounded-2xl md:rounded-[32px] overflow-hidden transition-all duration-500 ${expanded ? 'border-gray-300 ring-[4px] md:ring-[8px] ring-gray-50 shadow-xl' : 'hover:border-gray-300 border-gray-100 shadow-sm'}`}>
                        <div className="p-4 md:p-8 flex flex-col md:flex-row items-start md:items-center gap-4 md:gap-8 cursor-pointer bg-white" onClick={()=>setExpandedVideoId(expanded?null:record.id)}>
                          <div className="flex gap-4 md:contents w-full">
                            <img src={record.snippet.thumbnails.default.url} className="w-24 h-16 md:w-40 md:h-24 rounded-lg md:rounded-2xl object-cover border border-gray-100 shadow-sm shrink-0"/>
                            <div className="flex-1 overflow-hidden">
                              <h4 className="font-bold text-sm md:text-xl line-clamp-2 md:line-clamp-1 text-gray-900 tracking-tight mb-2 md:mb-3">{record.snippet.title}</h4>
                              <div className="flex flex-wrap items-center gap-1.5 md:gap-2">
                                <span className="text-[10px] md:text-xs font-black text-indigo-600 bg-indigo-50 border border-indigo-100 px-1.5 md:px-2 py-0.5 md:py-1 rounded-md">#{getMainKeyword(record)}</span>
                                <span className="text-[10px] md:text-xs font-black text-blue-600 bg-blue-50 px-2 md:px-2.5 py-0.5 md:py-1 rounded-md border border-blue-100">{record.snippet.channelTitle}</span>
                                <span className="text-[9px] md:text-[11px] font-bold text-gray-400 ml-1">업로드: {formatDate(record.snippet.publishedAt)}</span>
                              </div>
                            </div>
                          </div>
                          <div className="flex justify-between md:justify-end gap-4 md:gap-16 w-full md:w-auto text-left md:text-right pr-2 md:pr-6 items-center mt-2 md:mt-0 pt-2 md:pt-0 border-t md:border-none border-gray-50">
                            <div className="flex flex-row md:flex-col items-center md:items-end gap-2 md:gap-0"><p className="text-[10px] text-gray-400 font-bold md:mb-1">최종 누적 조회수</p><p className="font-black text-lg md:text-2xl text-gray-900 tracking-tighter">{formatNumber(latest.viewCount)}</p></div>
                            <div className={`p-2 md:p-4 bg-gray-50 rounded-xl md:rounded-2xl text-gray-400 transform transition-transform duration-500 border border-gray-100 ${expanded ? 'rotate-180 bg-gray-900 text-white border-gray-900' : ''}`}><ChevronDown className="w-5 h-5 md:w-6 md:h-6"/></div>
                          </div>
                        </div>
                        {expanded && (
                          <div className="bg-gray-50/50 p-4 md:p-10 border-t border-gray-100">
                            <div className="bg-white rounded-xl md:rounded-3xl border border-gray-100 overflow-x-auto shadow-sm">
                              <table className="w-full text-left min-w-[600px]">
                                <thead className="bg-gray-50/80 border-b border-gray-100">
                                  <tr>
                                    <th className="p-3 md:p-6 font-black text-[10px] md:text-[11px] text-gray-400 uppercase tracking-widest whitespace-nowrap">기록 일시</th>
                                    <th className="p-3 md:p-6 text-right font-black text-[10px] md:text-[11px] text-gray-400 uppercase tracking-widest whitespace-nowrap">누적 조회수</th>
                                    <th className="p-3 md:p-6 text-right font-black text-[10px] md:text-[11px] text-gray-400 uppercase tracking-widest whitespace-nowrap">누적 좋아요</th>
                                    <th className="p-3 md:p-6 text-right font-black text-[10px] md:text-[11px] text-gray-400 uppercase tracking-widest whitespace-nowrap">누적 댓글</th>
                                    <th className="p-3 md:p-6 text-right font-black text-[10px] md:text-[11px] text-gray-400 uppercase tracking-widest whitespace-nowrap">조회수 증가량</th>
                                  </tr>
                                </thead>
                                <tbody className="divide-y divide-gray-50">
                                  {keys.map((k, i) => {
                                    const data = record.history[k];
                                    const inc = i > 0 ? data.viewCount - record.history[keys[i-1]].viewCount : 0;
                                    return (
                                      <tr key={k} className="hover:bg-white transition-colors">
                                        <td className="p-3 md:p-6 font-bold text-xs md:text-sm text-gray-500 whitespace-nowrap">{k}</td>
                                        <td className="p-3 md:p-6 text-right font-black text-sm md:text-lg text-gray-900">{data.viewCount.toLocaleString()}</td>
                                        <td className="p-3 md:p-6 text-right font-bold text-xs md:text-sm text-gray-600">{data.likeCount.toLocaleString()}</td>
                                        <td className="p-3 md:p-6 text-right font-bold text-xs md:text-sm text-gray-600">{data.commentCount.toLocaleString()}</td>
                                        <td className="p-3 md:p-6 text-right">
                                          {i > 0 ? <span className="text-blue-600 font-black text-xs md:text-sm flex items-center justify-end gap-1"><ArrowUpRight size={14} className="md:w-4 md:h-4"/> {inc.toLocaleString()}</span> : <span className="text-gray-300 font-black text-sm md:text-lg">-</span>}
                                        </td>
                                      </tr>
                                    );
                                  })}
                                </tbody>
                              </table>
                            </div>
                          </div>
                        )}
                      </div>
                    );
                  })}
                </div>
              )}
            </div>
          )}
        </main>
      </div>

      {/* 모바일 하단 고정 네비게이션 */}
      <nav className="md:hidden fixed bottom-0 left-0 w-full bg-white border-t border-gray-200 flex justify-around items-center z-50 pb-safe">
        {tabsConfig.map(item => (
          <button 
            key={item.id} 
            onClick={() => setActiveTab(item.id)} 
            className={`flex flex-col items-center justify-center w-full py-3 text-[10px] font-bold transition-colors ${activeTab === item.id ? 'text-blue-600' : 'text-gray-400 hover:text-gray-600'}`}
          >
            {React.cloneElement(item.icon, { size: 22, className: `mb-1.5 ${activeTab === item.id ? 'fill-blue-50/50' : ''}` })}
            {item.label}
          </button>
        ))}
      </nav>
    </div>
  );
}
